178         function buildRow(any_gr,rowTemplate) {
179 
180             let result = {};
181 
182             function mergeCols(target,source_gr,template ) {
183 
184                 for(let fld in template) {
185                     
186                     //let elemOK = fld in source_gr;
187                     let elemOK = source_gr[fld] !== undefined;
188 
189                     if (!elemOK) {
190                         self._logInfo('{0} = {1}',fld,source_gr[fld]);
191 
192                         throw new Error(fld+' not found in '+elementTable(source_gr));
193                     }
194 
195                     if (template[fld] === true) {
196 
197                         target[fld] = source_gr[fld].getValue();
198 
199                     } else {
200 
201                         if (!target[fld]) target[fld] = {};
202 
203                         mergeCols(target[fld] , source_gr[fld] , template[fld]);
204                     }
205                 }
206             }
207

238             /**

244             function addColumn(dotfield,template) {
245                 let parts = dotfield.split('.');
246                 let last = parts.pop();
247                 let t = template;
248                 for(let part of parts) {

267         /**
268          * Apply the where object to the Glide object.
269          * It will be validated just before the query is executed
270          * as some fields might be DotWalkFieldName
271          * @param {GlideRecord} any_gr 
272          * @param {WhereList} whereObj 
273          */
274         function applyQuery(any_gr,whereObj) {
275 
276             function applyClause(clause) {
277 
278                 let eq1 = any_gr.getEncodedQuery();
279 
280                 if (false && typeof clause === 'string' ) {
281                     // Dont alow strings
282                     any_gr.addQuery(clause);
283                 } else {
284                     if (!Array.isArray(clause)) {
285 
286                         throw new Error('clause must be encodedString or array '+JSON.stringify(clause));
287                     }
288                     if (clause.length < 1 || clause.length > 3 ) {
289 
290                         throw new Error('Invalid clause '+JSON.stringify(clause));
291 
292                     } else if (clause.length === 2 && Array.isArray(clause[1])) {
293 
294                         any_gr.addQuery(clause[0],'IN',clause[1].join());
295 
296                     } else {
297 
298                         // Call addQuery
299                         any_gr.addQuery.apply(any_gr,clause);
300                     }
301                 }
302                 let eq2 = any_gr.getEncodedQuery();
303 
304                 if (eq1 === eq2) {
305                     throw new Error('Error adding clause '+clause);
306                 }
307             }
308 
309             function applyClauses(whereObj) {
310 
311                 for (let clause of whereObj) {
312 
313                     applyClause(clause);
314 
315                 }
316             }
317 
318             if (typeof whereObj === 'string' ) {
319                 applyClauses([whereObj]);
320             } else {
321                 applyClauses(whereObj);
322             }
323         }

375         /**
376          * If there are any nestedSelect items then flatten them
377          * eg. [ 
378          *  'sys_id' , 
379          *  { parent: [ 'sys_id', 'name' ] }
380 
381         * ] => [ 'sys_id' , 'parent.sys_id' , 'parent.name' ]
382         * 
383         * @param {SelectList} selectList 
384         */
385         function flattenSelectList(selectList) {
386 
387             function addCols(columns,prefix,cols) {
388                 for(let c of cols) {
389                     if (typeof c === 'string' ) {
390                         columns.push(prefix + c);
391                     } else {
392                         for(let prop in c) {
393                             addCols(prefix+prop+'.',c[prop]);
394                         }
395                     }
396                 }
397             }
398             
399             let columnsOut = [];
400 
401             if (typeof selectList === 'string' ) {
402 
403                 addCols(columnsOut,'',[selectList]);
404 
405             } else {
406 
407                 addCols(columnsOut,'',selectList);
408             }
409 
410             self._logInfo('flattenedList {0}',JSON.stringify(columnsOut));
411 
412             return columnsOut;

415         function addOrderBy(any_gr) {
416 
417             function addOrder(orderProp,grOrdeByMethod) {
418 
419                 let orderVal = queryDef[orderProp];
420 
421                 if (orderVal) {
422                     if (Array.isArray(orderVal)) {
423                         for(let fname of orderVal) {
424 
425                             self._logInfo('{0} {1}',orderProp,fname);
426                             any_gr[grOrdeByMethod](fname);
427                         }
428                     } else {
429                         self._logInfo('{0} {1}',orderProp,orderVal);
430                         any_gr[grOrdeByMethod](orderVal);
431 
432                     }
433                 }
434             }
435             addOrder('orderBy','orderBy');
436             addOrder('orderByDesc','orderByDesc');
437         }
438 
439         self._logInfo(JSON.stringify(queryDef));
440 
441         let any_gr = new GlideRecord(queryDef.from);
442 
443         if (!any_gr.isValid()) throw new Error('unknown table '+queryDef.from);
444 
445         applyQuery(any_gr,queryDef.where);
446 
447 
448         let flattenedSelectFields = flattenSelectList(queryDef.select);
449 
450         addReferencedFields(any_gr,flattenedSelectFields);
451 
452         addOrderBy(any_gr);
453 
454 
455         if (queryDef.limit) {
456             self._logInfo('limit {0}',queryDef.limit);
457             any_gr.setLimit(queryDef.limit);
458         }

461 
462         function t()  {
463             return new Date().getTime();
464         }
465         let times = {
466             start : t()
467         }
468         
469         let rowTemplate = buildRowTemplate(flattenedSelectFields);
470 
471         self.querySafe(any_gr);
472 
473         times.query = t() - times.start;
474 
475         while(any_gr.next()) {
476 
477             rows.push(buildRow(any_gr,rowTemplate));
478 
479         }
480         times.fetch = t() - times.start - times.query;
481 
482         self._logInfo(' times {0}',JSON.stringify(times));
483 
484         return rows;

631     // This function acts as a 'difference' method for arrays.
632     arrayDifference: function(arr1, arr2) {
633         // Create a Set from the second array for fast lookups.
634         const set2 = new Set(arr2);
635   
636         // Filter the first array, keeping only items NOT in the second Set.
637         return arr1.filter(item => !set2.has(item));
